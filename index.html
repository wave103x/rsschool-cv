void digitize(uintmax_t n, uint8_t digits[], size_t *length_out)
{
    if (n == 0)
    {
        *length_out = 1;
        digits[0] = 0;
    }
    else
    {
        uintmax_t tmp = n;
        int i = 0;
        while (tmp >= 1)
        {
            tmp /= 10;
            i++;
        }
        *length_out = i;

        for (uintmax_t j = 0; j < i; j++)
        {
            digits[j] = n % 10;
            n = (n - n % 10) / 10;
        }
    }
}
